def call_deepseek_v4_pro():
    ### 1. Grab the API key from the environment variables
    if not api_key:
    ### 2. Initialize the client using DeepSeek's official base URL
    client = OpenAI(
        api_key=api_key,
        base_url="https://api.deepseek.com"
    )
    print("Sending request to DeepSeek V4 Pro...")
    try:
        ### 3. Create the chat completion request
        response = client.chat.completions.create(
            model="deepseek-v4-pro",
            messages=[
                {"role": "system", "content": "You are a helpful and highly accurate assistant."},
                {"role": "user", "content": "Explain why a bicycle stays upright when moving forward, and do so in a concise format."}
            ],
            ### V4 models support explicit configuration for the reasoning engine
            extra_body={
                "thinking": {
                    "type": "enabled"
                },
                "reasoning_effort": "high"  ### Options: "high" or "max"
            },
            stream=False
        )
        ### 4. Extract and print the internal reasoning / thinking process (if available)
        message = response.choices[0].message
        if hasattr(message, 'reasoning_content') and message.reasoning_content:
            print("\n=== 🤔 DeepSeek Reasoning Trace ===")
            print(message.reasoning_content)
        ### 5. Extract and print the final answer
        print("\n=== ✨ Final Response ===")
        print(message.content)
    except Exception as e:
        print(f"\nAn error occurred while calling the API: {e}")

if __name__ == "__main__":
    call_deepseek_v4_pro()
